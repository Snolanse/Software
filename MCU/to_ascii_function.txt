void to_ascii(uint8_t str[], int num, int type)
{
	int i, rem, len = 0, n;
	n = num;
	//Finner antall siffer i ADC-verdien
	while (n != 0)
	{
		len++;
		n /= 10;
	}
	for (i = 0; i < len; i++)
	{
		rem = num % 10;
		num = num / 10;
		str[len - (i + 1)] = rem + '0';
	}
	
	str[len] = '\0';
	str[len+1] = '\n';
	str[len+2] = '\r';
	str_together(type, str);
	
}
